34{
35 GLenum error = glGetError();
36 if (error != GL_NO_ERROR)
37 {
38 switch (error)
39 {
40
41 case GL_INVALID_ENUM:
42 printf("GL Error: GL_INVALID_ENUM %s : %d \n", file, line);
43 break;
44 case GL_INVALID_VALUE:
45 printf("GL Error: GL_INVALID_VALUE %s : %d \n", file, line);
46 break;
